\doxysubsection{Detailed Description}
Fingerprint sensor driver (ZW111) for ESP32. 

\hypertarget{fingerprint_8h_autotoc_md0}{}\doxysubsubsection{\texorpdfstring{1. Hardware Interface Description}{1. Hardware Interface Description}}\label{fingerprint_8h_autotoc_md0}
\hypertarget{fingerprint_8h_autotoc_md1}{}\doxysubsubsubsection{\texorpdfstring{1.\+1 UART}{1.\+1 UART}}\label{fingerprint_8h_autotoc_md1}

\begin{DoxyItemize}
\item Default baud rate\+: 57600 bps (8 data bits, 1 stop bit, no parity)
\item Baud rate adjustable\+: 9600 to 115200 bps
\item Direct connection to MCU (3.\+3V) or use RS232 level converter for PC
\end{DoxyItemize}\hypertarget{fingerprint_8h_autotoc_md2}{}\doxysubsubsubsection{\texorpdfstring{1.\+2 Power-\/on Sequence\+:}{1.\+2 Power-\/on Sequence\+:}}\label{fingerprint_8h_autotoc_md2}

\begin{DoxyEnumerate}
\item Host receives fingerprint module (FPM) interrupt wake-\/up signal.
\item Host powers on Vmcu (MCU power supply) {\bfseries{before}} initializing UART.
\item After communication completes, pull down serial signal lines {\bfseries{before}} powering off Vmcu.
\end{DoxyEnumerate}\hypertarget{fingerprint_8h_autotoc_md3}{}\doxysubsubsubsection{\texorpdfstring{1.\+3 Connection}{1.\+3 Connection}}\label{fingerprint_8h_autotoc_md3}

\begin{DoxyItemize}
\item {\bfseries{TX}} → ESP32 GPIO17
\item {\bfseries{RX}} → ESP32 GPIO16
\item {\bfseries{VCC}} → 3.\+3V
\item {\bfseries{GND}} → GND 
\end{DoxyItemize}
